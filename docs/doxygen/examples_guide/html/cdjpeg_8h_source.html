121/*
122 * cjpeg/djpeg may need to perform extra passes to convert to or from
123 * the source/destination file format. The JPEG library does not know
124 * about these passes, but we'd like them to be counted by the progress
125 * monitor. We use an expanded progress monitor object to hold the
126 * additional pass count.
127 */
128
129struct cdjpeg_progress_mgr {
130 struct jpeg_progress_mgr pub; /* fields known to JPEG library */
131 int completed_extra_passes; /* extra passes completed */
132 int total_extra_passes; /* total extra */
133 /* last printed percentage stored here to avoid multiple printouts */
134 int percent_done;
135};
