23/* Structure for decoding tables. Each entry provides either the
24 information needed to do the operation requested by the code that
25 indexed that table entry, or it provides a pointer to another
26 table that indexes more bits of the code. op indicates whether
27 the entry is a pointer to another table, a literal, a length or
28 distance, an end-of-block, or an invalid code. For a table
29 pointer, the low four bits of op is the number of index bits of
30 that table. For a length or distance, the low four bits of op
31 is the number of extra bits to get after the code. bits is
32 the number of bits in this code or part of the code to drop off
33 of the bit buffer. val is the actual byte to output in the case
34 of a literal, the base length or distance, or the offset from
35 the current table to the next table. Each entry is four bytes. */
36typedef struct {
37 unsigned char op; /* operation, extra bits, table bits */
38 unsigned char bits; /* bits in this part of the code */
39 unsigned short val; /* offset in table or code value */
40} code;
41
42/* op values as set by inflate_table():
43 00000000 - literal
44 0000tttt - table link, tttt != 0 is the number of table index bits
45 0001eeee - length or distance, eeee is the number of extra bits
46 01100000 - end of block
47 01000000 - invalid code
48 */
49
50/* Maximum size of the dynamic table. The maximum number of code structures is
51 1444, which is the sum of 852 for literal/length codes and 592 for distance
52 codes. These values were found by exhaustive searches using the program
53 examples/enough.c found in the zlib distribtution. The arguments to that
54 program are the number of symbols, the initial root table size, and the
55 maximum bit length of a code. "enough 286 9 15" for literal/length codes
56 returns returns 852, and "enough 30 6 15" for distance codes returns 592.
57 The initial root table size (9 or 6) is found in the fifth argument of the
58 inflate_table() calls in inflate.c and infback.c. If the root table size is
59 changed, then these maximum sizes would be need to be recalculated and
60 updated. */
61#define ENOUGH_LENS 852
62#define ENOUGH_DISTS 592
63#define ENOUGH (ENOUGH_LENS+ENOUGH_DISTS)
