87/*
88 * These two functions are used to allocate and release small chunks of
89 * memory. (Typically the total amount requested through cjpeg_get_small is
90 * no more than 20K or so; this will be requested in chunks of a few K each.)
91 * Behavior should be the same as for the standard library functions th_malloc
92 * and th_free; in particular, cjpeg_get_small must return NULL on failure.
93 * On most systems, these ARE th_malloc and th_free. cjpeg_free_small is passed the
94 * size of the object being freed, just in case it's needed.
95 * On an 80x86 machine using small-data memory model, these manage near heap.
96 */
97
98EXTERN(void *) cjpeg_get_small JPP((j_common_ptr cinfo, size_t sizeofobject));
99EXTERN(void) cjpeg_free_small JPP((j_common_ptr cinfo, void * object,
100 size_t sizeofobject));
101
102/*
103 * These two functions are used to allocate and release large chunks of
104 * memory (up to the total th_free space designated by cjpeg_mem_available).
105 * The interface is the same as above, except that on an 80x86 machine,
106 * far pointers are used. On most other machines these are identical to
107 * the jpeg_get/free_small routines; but we keep them separate anyway,
108 * in case a different allocation strategy is desirable for large chunks.
109 */
110
111EXTERN(void FAR *) cjpeg_get_large JPP((j_common_ptr cinfo,
112 size_t sizeofobject));
113EXTERN(void) cjpeg_free_large JPP((j_common_ptr cinfo, void FAR * object,
114 size_t sizeofobject));

116/*
117 * The macro MAX_ALLOC_CHUNK designates the maximum number of bytes that may
118 * be requested in a single call to cjpeg_get_large (and cjpeg_get_small for that
119 * matter, but that case should never come into play). This macro is needed
120 * to model the 64Kb-segment-size limit of far addressing on 80x86 machines.
121 * On those machines, we expect that jconfig.h will provide a proper value.
122 * On machines with 32-bit flat address spaces, any large constant may be used.
123 *
124 * NB: jmemmgr.c expects that MAX_ALLOC_CHUNK will be representable as type
125 * size_t and will be a multiple of sizeof(align_type).
126 */
127
128#ifndef MAX_ALLOC_CHUNK /* may be overridden in jconfig.h */
129#define MAX_ALLOC_CHUNK 1000000000L
130#endif

132/*
133 * This routine computes the total space still available for allocation by
134 * cjpeg_get_large. If more space than this is needed, backing store will be
135 * used. NOTE: any memory already allocated must not be counted.
136 *
137 * There is a minimum space requirement, corresponding to the minimum
138 * feasible buffer sizes; jmemmgr.c will request that much space even if
139 * cjpeg_mem_available returns zero. The maximum space needed, enough to hold
140 * all working storage in memory, is also passed in case it is useful.
141 * Finally, the total space already allocated is passed. If no better
142 * method is available, cinfo->mem->max_memory_to_use - already_allocated
143 * is often a suitable calculation.
144 *
145 * It is OK for cjpeg_mem_available to underestimate the space available
146 * (that'll just lead to more backing-store access than is really necessary).
147 * However, an overestimate will lead to failure. Hence it's wise to subtract
148 * a slop factor from the true available space. 5% should be enough.
149 *
150 * On machines with lots of virtual memory, any large constant may be returned.
151 * Conversely, zero may be returned to always use the minimum amount of memory.
152 */
153
154EXTERN(long) cjpeg_mem_available JPP((j_common_ptr cinfo,
155 long min_bytes_needed,
156 long max_bytes_needed,
157 long already_allocated));
