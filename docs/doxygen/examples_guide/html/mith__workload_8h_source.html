29/* Structure: ee_work_item_t
30 This structure holds all of the information about a work item.
31
32 Static:
33 Information read in from the xml spec includes name, function pointers
34 and number of contexts.
35 Parameters to pass to the benchmark are also part of this structure.
36
37 Dynamic:
38 Information updated while the item is executed includes timing,
39 number of executions, failures etc.
40*/
41typedef struct ee_work_item_s {
42 char shortname[MITH_MAX_NAME]; /* a descriptor for the work item */
43 void *params; /* pointer to parameters, initialized by caller */
44 void * (*init_func)(void *params); /* init function, can be timed separately */
45 void * (*fini_func)(void *params); /* fini function, can be timed separately */
46 void * (*bench_func)(struct TCDef *,void *); /* actual benchmark execution (was t_run_test) */
47 int (*veri_func)(void *params); /* verify benchmark exec (return >0 if ok) */
48 int (*cleanup)(void *params); /* actual benchmark execution (was t_run_test) */
49 unsigned int num_contexts; /* number of contexts required */
50 size_t time; /* amount of time (cumulative) */
51 unsigned int finished; /* number of times finished execution */
52 unsigned int assigned; /* number of times assigned for execution */
53 unsigned int required; /* number of times required for execution */
54 unsigned int failed; /* number of times failed execution */
55 e_u32 uid; /* unique id for the work item in this workload (serial). */
56 e_u32 kernel_id; /* kernel id for the work item. common to all items using this kernel. */
57 e_u32 instance_id; /* instance id for the work item. common to all items with the same params. */
58 struct TCDef *tcdef; /* a way for the item to return info on run */
59 /* Support for synchronization */
60 e_s32 chain_id; /* Index in the chain (DFS number). -1 if no chains in workload */
61 e_s32 chain_uid; /* Each item is associated with a specific chain, identified by chain_uid */
62 e_u32 connect_idx[MAX_CONNECTIONS]; /* Mapping helpers to access actual connections */
63 e_s32 next_idx;
64} ee_work_item_t;

74/* Structure: ee_workload_s
75 This structure holds all the information about a workload
76 A workload structure is created from an xml workload definition.
77 The C file that creates and runs the workload is created using a perl
78 script found under util/perl/workload_parser.pl
79*/
80
81typedef struct ee_workload_s {
82 ee_pitem *load; /* array of all items in the workload */
83 unsigned int max_idx; /* number of items in workload */
84 unsigned int next_idx; /* next item to try and assign */
85 char shortname[MITH_MAX_NAME]; /* name for workload */
86 e_u32 uid; /* unique identifier for workload */
87 e_u32 iterations; /* number of iterations to run the workload */
88 unsigned int rev_M; /* revision of workload */
89 unsigned int rev_m; /* revision of workload */
90 /* SYNC related info */
91 struct ee_connection_s **connections;/* array of all connections active in the workload, by chain */
92 struct ee_connection_s *connection_protos;/* prototypes for all connections */
93 int num_connections; /* total number of connection prototypes */
94 int next_connect; /* for building connection prototypes */
95 unsigned int last_assign; /* id of last assign */
96 al_mutex_t **chain_mutexes; /* array containing mutexes for all chains, by chain */
97 chain_info_t *chain_info; /* array containing info on each chain, by chain */
98 int mcl; /* max chain length */
99 int num_chains; /* number of chains in item list (==#items/mcl) */
100 /* end of sync related info */
101} ee_workload;

137 Description:
138 main function that fires off work items.
139 Resides in <mith_lib.c>
140
141 Accepts a workload as a parameter, and number of iterations for the workload.
142
143 Will then fire off items from the workload to available contexts as soon as they become available.
144 Will stop once each item in the workload has iterated num_iterations, and report results.
145
146 Parameters:
147 workload - Actual workload to execute
148 num_iterations - Number of times to execute each item of the workload
149 num_contexts - number of execution contexts to execute on
150
151 Returns:
152 Instead of a return value, report results function is called to summarize results.
153 */
