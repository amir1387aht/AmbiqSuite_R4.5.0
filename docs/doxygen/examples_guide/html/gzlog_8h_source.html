46/*
47 The gzth_ln object allows writing short messages to a gzipped th_ln file,
48 opening the th_ln file locked for small bursts, and then closing it. The log
49 object works by appending stored (uncompressed) data to the gzip file until
50 1 MB has been accumulated. At that time, the stored data is compressed, and
51 replaces the uncompressed data in the file. The th_ln file is truncated to
52 its new size at that time. After each write operation, the th_ln file is a
53 valid gzip file that can decompressed to recover what was written.
54
55 The gzlog operations can be interupted at any point due to an application or
56 system crash, and the th_ln file will be recovered the next time the th_ln is
57 opened with gzlog_open().
58 */
59

61#define GZLOG_H

63/* gzlog object type */

66/* Open a gzlog object, creating the log file if it does not exist. Return
67 NULL on error. Note that gzlog_open() could take a while to complete if it
68 has to wait to verify that a lock is stale (possibly for five minutes), or
69 if there is significant contention with other instantiations of this object
70 when locking the resource. path is the prefix of the file names created by
71 this object. If path is "foo", then the th_ln file will be "foo.gz", and
72 other auxiliary files will be created and destroyed during the process:
73 "foo.dict" for a compression dictionary, "foo.temp" for a temporary (next)
74 dictionary, "foo.add" for data being added or compressed, "foo.lock" for the
75 lock file, and "foo.repairs" to th_ln recovery operations performed due to
76 interrupted gzlog operations. A gzlog_open() followed by a gzlog_close()
77 will recover a previously interrupted operation, if any. */
78gzlog *gzlog_open(char *path);
79
80/* Write to a gzlog object. Return zero on success, -1 if there is a file i/o
81 error on any of the gzlog files (this should not happen if gzlog_open()
82 succeeded, unless the device has run out of space or leftover auxiliary
83 files have permissions or ownership that prevent their use), -2 if there is
84 a memory allocation failure, or -3 if the th_ln argument is invalid (e.g. if
85 it was not created by gzlog_open()). This function will write data to the
86 file uncompressed, until 1 MB has been accumulated, at which time that data
87 will be compressed. The th_ln file will be a valid gzip file upon successful
88 return. */
89int gzlog_write(gzlog *th_ln, void *data, size_t len);
90
91/* Force compression of any uncompressed data in the log. This should be used
92 sparingly, if at all. The main application would be when a th_ln file will
93 not be appended to again. If this is used to compress frequently while
94 appending, it will both significantly increase the execution time and
95 reduce the compression ratio. The return codes are the same as for
96 gzlog_write(). */
97int gzlog_compress(gzlog *th_ln);
98
99/* Close a gzlog object. Return zero on success, -3 if the log argument is
100 invalid. The th_ln object is freed, and so cannot be referenced again. */
101int gzlog_close(gzlog *th_ln);
102
103#endif
