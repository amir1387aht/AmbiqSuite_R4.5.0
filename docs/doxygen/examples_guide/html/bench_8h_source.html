126static inline int
127CL_CHECK_SUBMIT(int i32ItemsPerCL)
128{
129 int _items_per_cl = i32ItemsPerCL;
130
131 if (_items_per_cl == 0)
132 {
133 _items_per_cl = ITEMS_PER_CL;
134 }
135
136 int items = 0;
137 int i32PixCount = 0;
138 start_wall = nema_get_time();
139 stop_wall = start_wall;
140
141 nema_cl_bind(g_psCLCur);
142 nema_cl_rewind(g_psCLCur);
143 nema_cl_branch(&g_sContextCL);
144
145 if (eExecMode != GPU_BOUND)
146 {
147 do
148 {
149 i32PixCount += i32RenderFrame();
150
151 ++items;
152 if ( (items % _items_per_cl) == 0)
153 {
154 if ( eExecMode != CPU_BOUND )
155 {
156 nema_cl_return();
157 nema_cl_submit(g_psCLCur);
158 PRINTF("Submitted CL %d\r\n", g_psCLCur->submission_id);
159 }
160
161 swap_cmd_lists();
162 PRINTF("Waiting for CL %d\r\n", g_psCLCur->submission_id);
163 nema_cl_wait(g_psCLCur);
164
165 nema_cl_bind(g_psCLCur);
166 nema_cl_rewind(g_psCLCur);
167 nema_cl_branch(&g_sContextCL);
168
169 stop_wall = nema_get_time();
170 if ((stop_wall-start_wall) > TIMEOUT_S)
171 {
172 break;
173 }
174 }
175 }
176 while (1);
177
178 PRINTF("Finishing Test\r\n");
179 PRINTF("Waiting for CL %d\r\n", g_psCLCur->submission_id);
180 nema_cl_wait(g_psCLCur);
181 swap_cmd_lists();
182
183 PRINTF("Waiting for CL %d\r\n", g_psCLCur->submission_id);
184 nema_cl_wait(g_psCLCur);
185 }
186 else
187 {
188 int i;
189 int pix_count_cl = 0;
190 for ( i = 0; i < _items_per_cl; i++ )
191 {
192 pix_count_cl += i32RenderFrame();
193 }
194 while (1)
195 {
196 nema_cl_return();
197 nema_cl_submit(g_psCLCur);
198 nema_cl_wait(g_psCLCur);
199 items += _items_per_cl;
200 i32PixCount += pix_count_cl;
201 stop_wall = nema_get_time();
202 if ((stop_wall-start_wall) > TIMEOUT_S)
203 {
204 break;
205 }
206
207 }
208 }
209
210 stop_wall = nema_get_time();
211
212 return i32PixCount;
213}
