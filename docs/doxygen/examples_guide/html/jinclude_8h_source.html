78/*
79 * We need the NULL macro and size_t typedef.
80 * On an ANSI-conforming system it is sufficient to include <stddef.h>.
81 * Otherwise, we get them from <stdlib.h> or <stdio.h>; we may have to
82 * pull in <sys/types.h> as well.
83 * Note that the core JPEG library does not require <stdio.h>;
84 * only the default error handler and data source/destination modules do.
85 * But we must pull it in because of the references to FILE in jpeglib.h.
86 * You can remove those references if you want to compile without <stdio.h>.
87 */
88
89/*
90 * We need memory copying and zeroing functions, plus strncpy().
91 * ANSI and System V implementations declare these in <string.h>.
92 * BSD doesn't have the mem() functions, but it does have bcopy()/bzero().
93 * Some systems may declare memset and memcpy in <memory.h>.
94 *
95 * NOTE: we assume the size parameters to these functions are of type size_t.
96 * Change the casts in these macros if not!
97 */
98
99#ifdef NEED_BSD_STRINGS
100
101#include <strings.h>
102#define MEMZERO(target,size) bzero((void *)(target), (size_t)(size))
103#define MEMCOPY(dest,src,size) bcopy((const void *)(src), (void *)(dest), (size_t)(size))
104
105#else /* not BSD, assume ANSI/SysV string lib */
106
107#include <string.h>
108#define MEMZERO(target,size) memset((void *)(target), 0, (size_t)(size))
109#define MEMCOPY(dest,src,size) memcpy((void *)(dest), (const void *)(src), (size_t)(size))
110
111#endif
112
113/*
114 * In ANSI C, and indeed any rational implementation, size_t is also the
115 * type returned by sizeof(). However, it seems there are some irrational
116 * implementations out there, in which sizeof() returns an int even though
117 * size_t is defined as long or unsigned long. To ensure consistent results
118 * we always use this SIZEOF() macro in place of using sizeof() directly.
119 */
120
121#define SIZEOF(object) ((size_t) sizeof(object))
122
123/*
124 * The modules that use fread() and fwrite() always invoke them through
125 * these macros. On some systems you may need to twiddle the argument casts.
126 * CAUTION: argument order is different from underlying functions!
127 */
128
129size_t cjpeg_fread(void *buf, size_t sizeofbuf, cjpparam_t *params);
130size_t cjpeg_fwrite(const void *buf, size_t sizeofbuf, cjpparam_t *params );

132#define JFREAD(file,buf,sizeofbuf) \

134#define JFWRITE(file,buf,sizeofbuf) \
