65/*
66 * A forward DCT routine is given a pointer to a work area of type DCTELEM[];
67 * the DCT is to be performed in-place in that buffer. Type DCTELEM is int
68 * for 8-bit samples, e_s32 for 12-bit samples. (NOTE: Floating-point DCT
69 * implementations use an array of type FAST_FLOAT, instead.)
70 * The DCT inputs are expected to be signed (range +-CENTERJSAMPLE).
71 * The DCT outputs are returned scaled up by a factor of 8; they therefore
72 * have a range of +-8K for 8-bit data, +-128K for 12-bit data. This
73 * convention improves accuracy in integer implementations and saves some
74 * work in floating-point ones.
75 * Quantization of the output coefficients is done by jcdctmgr.c.
76 */
77
78#if BITS_IN_JSAMPLE == 8
79typedef int DCTELEM; /* 16 or 32 bits is fine */
80#else
81typedef e_s32 DCTELEM; /* must have 32 bits */
82#endif
83
84typedef JMETHOD(void, forward_DCT_method_ptr, (DCTELEM * data));
85typedef JMETHOD(void, float_DCT_method_ptr, (FAST_FLOAT * data));

87/*
88 * An inverse DCT routine is given a pointer to the input JBLOCK and a pointer
89 * to an output sample array. The routine must dequantize the input data as
90 * well as perform the IDCT; for dequantization, it uses the multiplier table
91 * pointed to by compptr->dct_table. The output data is to be placed into the
92 * sample array starting at a specified column. (Any row offset needed will
93 * be applied to the array pointer before it is passed to the IDCT code.)
94 * Note that the number of samples emitted by the IDCT routine is
95 * DCT_scaled_size * DCT_scaled_size.
96 */
97
98/* typedef inverse_DCT_method_ptr is declared in jpegint.h */
99
100/*
101 * Each IDCT routine has its own ideas about the best dct_table element type.
102 */
103
104typedef MULTIPLIER ISLOW_MULT_TYPE; /* short or int, whichever is faster */
105#if BITS_IN_JSAMPLE == 8
106typedef MULTIPLIER IFAST_MULT_TYPE; /* 16 bits is OK, use short if faster */
107#define IFAST_SCALE_BITS 2 /* fractional bits in scale factors */
108#else
109typedef e_s32 IFAST_MULT_TYPE; /* need 32 bits for scaled quantizers */
110#define IFAST_SCALE_BITS 13 /* fractional bits in scale factors */
111#endif
112typedef FAST_FLOAT FLOAT_MULT_TYPE; /* preferred floating type */

114/*
115 * Each IDCT routine is responsible for range-limiting its results and
116 * converting them to unsigned form (0..MAXJSAMPLE). The raw outputs could
117 * be quite far out of range if the input data is corrupt, so a bulletproof
118 * range-limiting step is required. We use a mask-and-table-lookup method
119 * to do the combined operations quickly. See the comments with
120 * prepare_range_limit_table (in jdmaster.c) for more info.
121 */
122
123#define IDCT_range_limit(cinfo) ((cinfo)->sample_range_limit + CENTERJSAMPLE)
124
125#define RANGE_MASK (MAXJSAMPLE * 4 + 3) /* 2 bits wider than legal samples */

166/*
167 * Macros for handling fixed-point arithmetic; these are used by many
168 * but not all of the DCT/IDCT modules.
169 *
170 * All values are expected to be of type e_s32.
171 * Fractional constants are scaled left by CONST_BITS bits.
172 * CONST_BITS is defined within each module using these macros,
173 * and may differ from one module to the next.
174 */
175
176#define ONE ((e_s32) 1)
177#define CONST_SCALE (ONE << CONST_BITS)
178
179/* Convert a positive real constant to an integer scaled by CONST_SCALE.
180 * Caution: some C compilers fail to reduce "FIX(constant)" at compile time,
181 * thus causing a lot of useless floating-point operations at run time.
182 */
183
184#define FIX(x) ((e_s32) ((x) * CONST_SCALE + 0.5))
185
186/* Descale and correctly round an e_s32 value that's scaled by N bits.
187 * We assume RIGHT_SHIFT rounds towards minus infinity, so adding
188 * the fudge factor is correct for either sign of X.
189 */
190
191#define DESCALE(x,n) RIGHT_SHIFT((x) + (ONE << ((n)-1)), n)
192
193/* Multiply an e_s32 variable by an e_s32 constant to yield an e_s32 result.
194 * This macro is used only when the two inputs will actually be no more than
195 * 16 bits wide, so that a 16x16->32 bit multiply can be used instead of a
196 * full 32x32 multiply. This provides a useful speedup on many machines.
197 * Unfortunately there is no way to specify a 16x16->32 multiply portably
198 * in C, but some C compilers will do the right thing if you provide the
199 * correct combination of casts.
200 */
201
202#ifdef SHORTxSHORT_32 /* may work if 'int' is 32 bits */
203#define MULTIPLY16C16(var,constant) (((e_s16) (var)) * ((e_s16) (constant)))
204#endif
205#ifdef SHORTxLCONST_32 /* known to work with Microsoft C 6.0 */
206#define MULTIPLY16C16(var,constant) (((e_s16) (var)) * ((e_s32) (constant)))
207#endif
208
209#ifndef MULTIPLY16C16 /* default definition */
210#define MULTIPLY16C16(var,constant) ((var) * (constant))
211#endif
212
213/* Same except both inputs are variables. */
214
215#ifdef SHORTxSHORT_32 /* may work if 'int' is 32 bits */
216#define MULTIPLY16V16(var1,var2) (((e_s16) (var1)) * ((e_s16) (var2)))
217#endif
218
219#ifndef MULTIPLY16V16 /* default definition */
220#define MULTIPLY16V16(var1,var2) ((var1) * (var2))
