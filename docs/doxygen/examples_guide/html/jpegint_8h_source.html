317/* We assume that right shift corresponds to signed division by 2 with
318 * rounding towards minus infinity. This is correct for typical "arithmetic
319 * shift" instructions that shift in copies of the sign bit. But some
320 * C compilers implement >> with an unsigned shift. For these machines you
321 * must define RIGHT_SHIFT_IS_UNSIGNED.
322 * RIGHT_SHIFT provides a proper signed right shift of an e_s32 quantity.
323 * It is only applied with constant shift counts. SHIFT_TEMPS must be
324 * included in the variables of any routine using RIGHT_SHIFT.
325 */
326
327#ifdef RIGHT_SHIFT_IS_UNSIGNED
328#define SHIFT_TEMPS e_s32 shift_temp;
329#define RIGHT_SHIFT(x,shft) \
330 ((shift_temp = (x)) < 0 ? \
331 (shift_temp >> (shft)) | ((~((e_s32) 0)) << (32-(shft))) : \
332 (shift_temp >> (shft)))
333#else
334#define SHIFT_TEMPS
335#define RIGHT_SHIFT(x,shft) ((x) >> (shft))
336#endif
