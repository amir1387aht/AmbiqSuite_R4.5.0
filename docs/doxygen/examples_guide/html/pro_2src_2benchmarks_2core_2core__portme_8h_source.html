128/* Configuration: MULTITHREAD
129 Define for parallel execution
130
131 Valid values:
132 1 - only one context (default).
133 N>1 - will execute N copies in parallel.
134
135 Note:
136 If this flag is defined to more then 1, an implementation for launching parallel contexts must be defined.
137
138 Two sample implementations are provided. Use <USE_PTHREAD> or <USE_FORK> to enable them.
139
140 It is valid to have a different implementation of <core_start_parallel> and <core_end_parallel> in <core_portme.c>,
141 to fit a particular architecture.
142*/
143#ifndef MULTITHREAD
144#define MULTITHREAD 1
145#endif

147/* Configuration: USE_PTHREAD
