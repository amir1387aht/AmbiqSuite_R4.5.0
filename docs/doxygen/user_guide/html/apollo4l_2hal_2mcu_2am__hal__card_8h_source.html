516 const uint32_t __mask = (__size < 32 ? 1 << __size : 0) - 1;
517 const int __off = 3 - ((start) / 32);
518 const int __shft = (start)&31;
519 uint32_t __res;
520
521 __res = resp[__off] >> __shft;
522 if (__size + __shft > 32)
523 {
524 __res |= resp[__off - 1] << ((32 - __shft) % 32);
525 }
526 return __res &__mask;
527}

529static inline uint32_t am_hal_unstuff_bytes(uint32_t *ext_csd, uint32_t start, uint32_t size)
530{
531 int i;
532 const uint32_t __size = size;
533 uint32_t __res = 0x0;
534 const uint8_t *__p = (const uint8_t *)ext_csd;
535 for (i = 0; i < __size; i++)
536 {
537 __res |= __p[start + i] << 8 * i;
538 }
539 return __res;
540}
